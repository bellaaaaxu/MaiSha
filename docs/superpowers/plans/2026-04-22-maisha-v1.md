# 买啥 MaiSha v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 交付可上线的微信小程序"买啥 MaiSha" v1，两用户共享购物清单，按超市分组、按品类副分组、实时同步、分享邀请。

**Architecture:** 微信小程序原生（TypeScript）+ Vant Weapp 组件库 + wxcloud 云开发后端（云数据库 + 云函数）。核心业务逻辑提取到纯函数 utils（jest 可测）；UI 由 pages + components 组合。数据流：单一 `lists` 文档 + 多 `items` 文档，两端通过 `db.watch` 实时订阅。

**Tech Stack:** WeChat Mini Program + TypeScript + Vant Weapp + wxcloud (DB + Cloud Functions) + Jest (unit tests) + WeChat DevTools (Windows IDE)

**参考:** [docs/superpowers/specs/2026-04-22-maisha-design.md](../specs/2026-04-22-maisha-design.md)

```typescript
import { CATEGORY_DEFS, FALLBACK_CATEGORY, FALLBACK_CATEGORY_EMOJI } from './constants';
import type { CategoryKey } from '@/types/item';

export interface MatchResult {
  category: CategoryKey;
  emoji: string;
}

export function matchCategory(name: string): MatchResult {
  if (!name) {
    return { category: FALLBACK_CATEGORY, emoji: FALLBACK_CATEGORY_EMOJI };
  }
  for (const def of CATEGORY_DEFS) {
    for (const kw of def.keywords) {
      if (name.includes(kw)) {
        return { category: def.key, emoji: def.emoji };
      }
    }
  }
  return { category: FALLBACK_CATEGORY, emoji: FALLBACK_CATEGORY_EMOJI };
}
```

```typescript
export interface FrequentItem {
  name: string;
  note: string;
  supermarket: string;
  categoryEmoji: string;
  count: number;
  lastUsedAt: number;
}

export type UsageInput = Pick<FrequentItem, 'name' | 'note' | 'supermarket' | 'categoryEmoji'>;

function storageKey(openid: string): string {
  return `frequent_${openid}`;
}

function aggregationKey(i: UsageInput): string {
  return `${i.name}|${i.note}|${i.supermarket}`;
}

function load(openid: string): FrequentItem[] {
  const raw = wx.getStorageSync(storageKey(openid));
  if (!raw || !Array.isArray(raw)) return [];
  return raw as FrequentItem[];
}

function save(openid: string, items: FrequentItem[]): void {
  wx.setStorageSync(storageKey(openid), items);
}

export function recordItemUsage(openid: string, input: UsageInput): void {
  const all = load(openid);
  const key = aggregationKey(input);
  const existing = all.find(it => aggregationKey(it) === key);
  const now = Date.now();
  if (existing) {
    existing.count += 1;
    existing.lastUsedAt = now;
  } else {
    all.push({ ...input, count: 1, lastUsedAt: now });
  }
  save(openid, all);
}

export function getTopFrequentItems(openid: string, limit: number): FrequentItem[] {
  const all = load(openid);
  return all
    .slice()
    .sort((a, b) => b.count - a.count || b.lastUsedAt - a.lastUsedAt)
    .slice(0, limit);
}
```

```typescript
import type { Item, CategoryKey } from '@/types/item';
import type { Supermarket } from '@/types/supermarket';
import { UNDELETABLE_SUPERMARKET_ID } from './constants';

export interface CategoryGroup {
  category: CategoryKey;
  emoji: string;
  items: Item[];
}

export interface MarketGroup {
  supermarket: Supermarket;
  categories: CategoryGroup[];
  totalCount: number;
}

export function groupItemsByMarketAndCategory(
  items: Item[],
  supermarkets: Supermarket[]
): MarketGroup[] {
  if (!items.length) return [];

  const validIds = new Set(supermarkets.map(s => s.id));
  const fallbackId = UNDELETABLE_SUPERMARKET_ID;
  const marketMap = new Map<string, Supermarket>(supermarkets.map(s => [s.id, s]));

  // Bucket items by market id
  const byMarket = new Map<string, Item[]>();
  for (const item of items) {
    const mid = validIds.has(item.supermarket) ? item.supermarket : fallbackId;
    if (!byMarket.has(mid)) byMarket.set(mid, []);
    byMarket.get(mid)!.push(item);
  }

  // Build groups in supermarkets order (only non-empty)
  const result: MarketGroup[] = [];
  for (const s of supermarkets) {
    const bucket = byMarket.get(s.id);
    if (!bucket || !bucket.length) continue;

    // Sub-group by category, preserve first-appearance order
    const catOrder: CategoryKey[] = [];
    const catMap = new Map<CategoryKey, CategoryGroup>();
    for (const item of bucket) {
      if (!catMap.has(item.category)) {
        catOrder.push(item.category);
        catMap.set(item.category, {
          category: item.category,
          emoji: item.categoryEmoji,
          items: []
        });
      }
      catMap.get(item.category)!.items.push(item);
    }

    result.push({
      supermarket: marketMap.get(s.id)!,
      categories: catOrder.map(k => catMap.get(k)!),
      totalCount: bucket.length
    });
  }

  return result;
}
```

```typescript
import type { Item } from '@/types/item';
import type { Supermarket } from '@/types/supermarket';
import { groupItemsByMarketAndCategory } from './group-items';

function fmtDate(d: Date): string {
  return `${d.getMonth() + 1}/${d.getDate()}`;
}

function fmtItem(it: Item): string {
  const parts = [it.name];
  if (it.note) parts.push(it.note);
  if (it.quantity) parts.push(`× ${it.quantity}`);
  return parts.join(' · ');
}

export function generateShareText(
  items: Item[],
  supermarkets: Supermarket[],
  now: Date = new Date()
): string {
  const unchecked = items.filter(i => !i.checked);
  if (!unchecked.length) {
    return `🛒 买啥 · ${fmtDate(now)}\n\n清单为空 ✨`;
  }

  const groups = groupItemsByMarketAndCategory(unchecked, supermarkets);
  const lines: string[] = [`🛒 买啥 · ${fmtDate(now)} 购物清单`, ''];
  for (const g of groups) {
    lines.push(`${g.supermarket.emoji} ${g.supermarket.name}`);
    for (const c of g.categories) {
      for (const it of c.items) {
        lines.push(`  • ${fmtItem(it)}`);
      }
    }
    lines.push('');
  }
  return lines.join('\n').trimEnd();
}
```

```javascript
const db = cloud.database();

const DEFAULT_SUPERMARKETS = [
  { id: 'hm',   name: '盒马',     emoji: '🛒' },
  { id: 'cc',   name: '菜场',     emoji: '🥬' },
  { id: 'sm',   name: '山姆',     emoji: '🏬' },
  { id: 'jd',   name: '京东到家', emoji: '📦' },
  { id: 'cvs',  name: '便利店',   emoji: '🏪' },
  { id: 'none', name: '未分类',   emoji: '❓' }
];

/**
 * event.joinListId?: string — 从分享卡片 scene 带入的 listId
 *
 * Behavior:
 *   1) 若 joinListId 存在且该 list 存在，把 openid 加入 memberOpenids，返回该 list
 *   2) 否则查找 openid 为 memberOpenids 成员的 list；找到返回
 *   3) 否则创建新 list（openid 为 owner），返回
 */
exports.main = async (event) => {
  const { OPENID } = cloud.getWXContext();
  if (!OPENID) return { ok: false, error: 'no-openid' };

  const now = Date.now();
  const { joinListId } = event || {};

  // 1) join a shared list
  if (joinListId) {
    const { data: list } = await db.collection('lists').doc(joinListId).get().catch(() => ({ data: null }));
    if (list) {
      if (!list.memberOpenids.includes(OPENID)) {
        await db.collection('lists').doc(joinListId).update({
          data: {
            memberOpenids: db.command.addToSet(OPENID),
            updatedAt: now
          }
        });
        list.memberOpenids.push(OPENID);
      }
      return { ok: true, list };
    }
  }

  // 2) find existing list the user belongs to
  const found = await db.collection('lists')
    .where({ memberOpenids: OPENID })
    .limit(1)
    .get();
  if (found.data.length) {
    return { ok: true, list: found.data[0] };
  }

  // 3) create new list
  const created = await db.collection('lists').add({
    data: {
      name: '家里',
      ownerOpenid: OPENID,
      memberOpenids: [OPENID],
      supermarkets: DEFAULT_SUPERMARKETS,
      createdAt: now,
      updatedAt: now
    }
  });

  const { data: list } = await db.collection('lists').doc(created._id).get();
  return { ok: true, list };
};
```

```typescript
    // Else wait for app.ts to call onListReady
  },

  onListReady(list) {
    this.setData({ list });
    this.startWatch(list._id);
  },

  async startWatch(listId) {
    // Initial load
    try {
      const items = await fetchItems(listId);
      this.setData({ items, loading: false });
      this.refreshGroups();
    } catch (err) {
      console.error('fetchItems failed', err);
      this.setData({ loading: false });
      wx.showToast({ title: '加载失败', icon: 'none' });
    }

    // Real-time watch
    this.watcher = watchItems(
      listId,
      items => {
        this.setData({ items });
        this.refreshGroups();
      },
      err => {
        console.error('watchItems error', err);
        wx.showToast({ title: '同步断开，重试中', icon: 'none' });
      }
    );
  },

  onUnload() {
    if (this.watcher) this.watcher.close();
  },

  refreshGroups() {
    const list = this.data.list;
    if (!list) return;
    const groups = groupItemsByMarketAndCategory(this.data.items, list.supermarkets);
    const unchecked = this.data.items.filter(i => !i.checked).length;
    this.setData({ groups, uncheckedCount: unchecked });
  },

  onItemTap(_e) {
    wx.showToast({ title: '（下一步接入）勾选', icon: 'none' });
  },

  onItemLongPress(_e) {
    wx.showToast({ title: '（下一步接入）编辑/删除', icon: 'none' });
  },

  onAdd() {
    wx.showToast({ title: '（下一步接入）添加', icon: 'none' });
  },

  onShare() {
    wx.showToast({ title: '（下一步接入）分享', icon: 'none' });
  },

  onMoreMenu() {
    wx.showToast({ title: '（下一步接入）更多', icon: 'none' });
  }
});
```

```typescript
import type { FrequentItem } from '@/utils/frequent-items';
import { getTopFrequentItems } from '@/utils/frequent-items';
import { matchCategory } from '@/utils/category-matcher';
import type { NewItemInput } from '@/types/item';

interface ChipVM {
  key: string;
  label: string;
  categoryEmoji: string;
  name: string;
  note: string;
  supermarket: string;
  category: string;
}

Component({
  properties: {
    show: { type: Boolean, value: false },
    openid: { type: String, value: '' }
  },
  data: {
    inputValue: '',
    frequentChips: [] as ChipVM[],
    recentChips: [] as ChipVM[]
  },
  observers: {
    'show'(next: boolean) {
      if (next) this.loadChips();
      else this.setData({ inputValue: '' });
    }
  },
  methods: {
    loadChips() {
      const openid = this.data.openid;
      if (!openid) return;
      const freq: FrequentItem[] = getTopFrequentItems(openid, 6);
      // "最近添加" = 按 lastUsedAt desc 取前 5（不考虑 count）
      const allSorted = getTopFrequentItems(openid, 9999);
      const recent = allSorted
        .slice()
        .sort((a, b) => b.lastUsedAt - a.lastUsedAt)
        .slice(0, 5);

      this.setData({
        frequentChips: freq.map(this.toVM),
        recentChips: recent.map(this.toVM)
      });
    },
    toVM(f: FrequentItem): ChipVM {
      const label = f.note ? `${f.name} ${f.note}` : f.name;
      const match = matchCategory(f.name);
      return {
        key: `${f.name}|${f.note}|${f.supermarket}`,
        label,
        categoryEmoji: f.categoryEmoji || match.emoji,
        name: f.name,
        note: f.note,
        supermarket: f.supermarket,
        category: match.category
      };
    },
    onInput(e: WechatMiniprogram.Input) {
      this.setData({ inputValue: e.detail.value });
    },
    onSubmit() {
      const name = this.data.inputValue.trim();
      if (!name) return;
      const match = matchCategory(name);
      const input: NewItemInput = {
        name,
        note: '',
        quantity: '',
        supermarket: 'none',
        category: match.category,
        categoryEmoji: match.emoji
      };
      this.triggerEvent('submit', { input });
      this.setData({ inputValue: '' });
    },
    onChipTap(e: WechatMiniprogram.CustomEvent) {
      const vm = e.detail.payload as ChipVM;
      const input: NewItemInput = {
        name: vm.name,
        note: vm.note,
        supermarket: vm.supermarket,
        category: vm.category as any,
        categoryEmoji: vm.categoryEmoji
      };
      this.triggerEvent('submit', { input });
    },
    onClose() {
      this.triggerEvent('close');
    },
    noop() { /* stop event */ }
  }
});
```

```typescript
import type { Item, CategoryKey } from '@/types/item';
import type { Supermarket } from '@/types/supermarket';
import { updateItem, deleteItem } from '@/utils/db';
import { CATEGORY_DEFS, FALLBACK_CATEGORY, FALLBACK_CATEGORY_EMOJI } from '@/utils/constants';

interface PageData {
  loading: boolean;
  itemId: string;
  form: { name: string; note: string; quantity: string };
  supermarkets: Supermarket[];
  marketOptions: { id: string; label: string }[];
  marketIndex: number;
  marketLabel: string;
  categoryOptions: { key: CategoryKey; emoji: string; label: string }[];
  categoryIndex: number;
  categoryLabel: string;
  pickedEmoji: string;
}

Page<PageData, any>({
  data: {
    loading: true,
    itemId: '',
    form: { name: '', note: '', quantity: '' },
    supermarkets: [],
    marketOptions: [],
    marketIndex: 0,
    marketLabel: '',
    categoryOptions: [],
    categoryIndex: 0,
    categoryLabel: '',
    pickedEmoji: ''
  },

  async onLoad(opts: { id?: string }) {
    const app = getApp<IAppOption>();
    const list = app.globalData.list;
    if (!list || !opts.id) {
      wx.showToast({ title: '参数错误', icon: 'none' });
      wx.navigateBack();
      return;
    }

    const db = wx.cloud.database();
    const { data: item } = await db.collection('items').doc(opts.id).get() as { data: Item };

    const marketOptions = list.supermarkets.map(s => ({ id: s.id, label: `${s.emoji} ${s.name}` }));
    const marketIndex = Math.max(0, marketOptions.findIndex(o => o.id === item.supermarket));
    const marketLabel = marketOptions[marketIndex].label;

    const categoryOptions = [
      ...CATEGORY_DEFS.map(d => ({ key: d.key, emoji: d.emoji, label: `${d.emoji} ${d.key}` })),
      { key: FALLBACK_CATEGORY, emoji: FALLBACK_CATEGORY_EMOJI, label: `${FALLBACK_CATEGORY_EMOJI} ${FALLBACK_CATEGORY}` }
    ];
    const categoryIndex = Math.max(0, categoryOptions.findIndex(o => o.key === item.category));
    const categoryLabel = categoryOptions[categoryIndex].label;

    this.setData({
      loading: false,
      itemId: item._id,
      form: { name: item.name, note: item.note || '', quantity: item.quantity || '' },
      supermarkets: list.supermarkets,
      marketOptions, marketIndex, marketLabel,
      categoryOptions, categoryIndex, categoryLabel,
      pickedEmoji: item.categoryEmoji
    });
  },

  onName(e: WechatMiniprogram.Input)     { this.setData({ 'form.name':     e.detail.value }); },
  onNote(e: WechatMiniprogram.Input)     { this.setData({ 'form.note':     e.detail.value }); },

    const idx = Number(e.detail.value);
    this.setData({ marketIndex: idx, marketLabel: this.data.marketOptions[idx].label });
  },

  onCategoryPick(e: WechatMiniprogram.PickerChange) {
    const idx = Number(e.detail.value);
    const opt = this.data.categoryOptions[idx];
    this.setData({
      categoryIndex: idx,
      categoryLabel: opt.label,
      pickedEmoji: opt.emoji
    });
  },

  async onSave() {
    const d = this.data;
    if (!d.form.name.trim()) {
      wx.showToast({ title: '名称不能为空', icon: 'none' });
      return;
    }
    try {
      await updateItem(d.itemId, {
        name: d.form.name.trim(),
        note: d.form.note,
        quantity: d.form.quantity,
        supermarket: d.marketOptions[d.marketIndex].id,
        category: d.categoryOptions[d.categoryIndex].key,
        categoryEmoji: d.pickedEmoji
      });
      wx.showToast({ title: '已保存', icon: 'success' });

    if (!confirm.confirm) return;
    try {
      await deleteItem(this.data.itemId);

      setTimeout(() => wx.navigateBack(), 500);
    } catch {
      wx.showToast({ title: '删除失败', icon: 'none' });
    }
  }
```

```typescript
import type { Supermarket } from '@/types/supermarket';
import { updateListSupermarkets } from '@/utils/db';

Page<any, any>({
  data: {
    supermarkets: [] as Supermarket[],
    newName: '',
    newEmoji: ''
  },

  onLoad() {
    const app = getApp<IAppOption>();
    if (!app.globalData.list) return;
    this.setData({ supermarkets: JSON.parse(JSON.stringify(app.globalData.list.supermarkets)) });
  },

  onNameInput(e: WechatMiniprogram.Input & { currentTarget: { dataset: { index: number } } }) {
    const idx = Number(e.currentTarget.dataset.index);
    const supermarkets = this.data.supermarkets.slice();
    supermarkets[idx] = { ...supermarkets[idx], name: e.detail.value };
    this.setData({ supermarkets });
  },

  onRemove(e: WechatMiniprogram.TouchEvent & { currentTarget: { dataset: { index: number } } }) {
    const idx = Number(e.currentTarget.dataset.index);
    if (this.data.supermarkets[idx].id === UNDELETABLE_SUPERMARKET_ID) {
      wx.showToast({ title: '"未分类"不可删', icon: 'none' });
      return;
    }
    const supermarkets = this.data.supermarkets.slice();
    supermarkets.splice(idx, 1);
    this.setData({ supermarkets });
  },

  onNewName(e: WechatMiniprogram.Input)  { this.setData({ newName:  e.detail.value }); },
  onNewEmoji(e: WechatMiniprogram.Input) { this.setData({ newEmoji: e.detail.value }); },

  onAdd() {
    const { newName, newEmoji, supermarkets } = this.data;
    if (!newName.trim()) return;
    const id = 'sm_' + Date.now().toString(36);
    supermarkets.push({ id, name: newName.trim(), emoji: newEmoji.trim() || '🏪' });
    this.setData({ supermarkets, newName: '', newEmoji: '' });
  },

  async onSave() {
    const app = getApp<IAppOption>();
    const list = app.globalData.list;
    if (!list) return;
    try {
      await updateListSupermarkets(list._id, this.data.supermarkets);
      // update local
      list.supermarkets = this.data.supermarkets;
      wx.showToast({ title: '已保存', icon: 'success' });
      setTimeout(() => wx.navigateBack(), 500);
    } catch {
      wx.showToast({ title: '保存失败', icon: 'none' });
    }
  }
});
```
